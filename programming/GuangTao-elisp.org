(add-hook 'org-mode-hook (lambda ()
  ;;……
  (cond
   ;;如果文件最后8个字符是data.org
     ((equal (substring buffer-file-name -8) "data.org")
       (progn ;;这是只对data.org文件生效的快捷键
          (local-set-key (kbd "C") 'next-line)
       ))
     (t ;对不是data.org的org文件生效的快捷键
          (local-set-key (kbd "C") 'hydra-org-mode/body)
     ))
))
